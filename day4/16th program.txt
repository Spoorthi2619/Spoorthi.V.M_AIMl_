//write a c program to print the following pattern 

#include<stdio.h>

int main()

{
    
int num;
    
printf("enter the number:\n");
    
scanf("%d",&num);
    
for(int i=1;i<=num;i++)
    
{
        
for(int j=1;j<=i;-j++)
        
{
            
printf("%c ",(64+j));
            
        
}
         
printf("\n");
    
} 

}
