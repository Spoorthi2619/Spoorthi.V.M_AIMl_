// write a c program to read the input for num as integer and print the numbers from 1 to num

#include<stdio.h>

int main()

{
    
int num;
    
printf("enter the number:");
    
scanf("%d \n",&num);
    
//for loop for iteration
    
for(int i=1;i<=num;i++)
    
{
       
printf("%d \n",i);
    
}

}
